# NixOS Installation Guide (Donkeykong)
# Disko declarative partitioning with LUKS encryption

## Prerequisites
- Boot from NixOS live USB
- Open terminal, run as root/sudo
- 1Password account credentials handy (email, secret key, password) --
  needed for `op signin` on the live USB. There's no other path; git-crypt
  was retired in #86.
- An existing nixerator host (the source machine) reachable for scp of
  SSH/GPG keys. The 1Password SA token is fetched directly from your
  Personal vault via `op read` -- not scp'd.

## Variables (set once)

# bash (for fish, use `set -gx` instead of `export`)
export TARGET_HOST="donkeykong"
export INSTALL_USER="<your-local-username>"       # must match globals.nix user.name
export SOURCE_USER="<user-on-existing-machine>"
export SOURCE_HOST="<existing-machine-ip>"
export SOURCE_SSH="$SOURCE_USER@$SOURCE_HOST"
export REPO_SSH_URL="git@github.com:bashfulrobot/nixerator.git"
export TARGET_HOME="/mnt/home/$INSTALL_USER"
export TARGET_REPO_DIR="$TARGET_HOME/dev/nix"
export TARGET_REPO_PATH="$TARGET_REPO_DIR/nixerator"

## Step 1: Verify Internet
ping -c 3 google.com

## Step 2: Retrieve SSH Keys (from source machine)
mkdir -p ~/.ssh && chmod 700 ~/.ssh
scp "$SOURCE_SSH:~/.ssh/id_*" ~/.ssh/ 2>/dev/null || true
chmod 600 ~/.ssh/id_* 2>/dev/null || true
scp "$SOURCE_SSH:~/.ssh/id_*.pub" ~/.ssh/ 2>/dev/null || true
chmod 644 ~/.ssh/id_*.pub 2>/dev/null || true

## Step 3: Clone Repository
mkdir -p /tmp/install && cd /tmp/install
nix-shell -p git _1password-cli --run "git clone $REPO_SSH_URL"
cd nixerator

## Step 4: Stage secrets (1Password) for nixos-install
# The flake reads ~/.config/nixos-secrets/secrets.json at eval time. On a
# fresh live USB we need to render that file BEFORE nixos-install runs.
# The bootstrap-install-secrets.sh helper walks you through it interactively:
#   - Verifies op CLI signed in (prompts for `op signin` if not)
#   - Fetches the SA token from your Personal vault via `op read` (one biometric)
#   - Installs token at  /home/$INSTALL_USER/.config/op/service-account-token (0600)
#   - Renders secrets at /home/$INSTALL_USER/.config/nixos-secrets/secrets.json (0600)
sudo INSTALL_USER="$INSTALL_USER" ./extras/helpers/bootstrap-install-secrets.sh stage

## Step 5: Verify Disk
ls -l /dev/disk/by-id/ | grep nvme
# Must match disk ID in hosts/donkeykong/disko.nix
# Edit disko.nix if different

## Step 6: Partition + Format (ERASES ALL DATA)
sudo nix --experimental-features "nix-command flakes" run github:nix-community/disko -- \
  --mode disko --flake ".#$TARGET_HOST"
# Enter LUKS password when prompted -- remember it!

## Step 7: Generate Hardware Config
sudo nixos-generate-config --root /mnt --no-filesystems
sudo cp /mnt/etc/nixos/hardware-configuration.nix ./hosts/donkeykong/
git add hosts/donkeykong/hardware-configuration.nix
git commit -m "feat(donkeykong): add hardware configuration"

## Step 8: Install NixOS
# --impure is required: flake.nix reads secretsFile via builtins.readFile of
# an absolute string path. Required since the secrets cached file lives
# outside the repo (in $INSTALL_USER's home).
sudo nixos-install --impure --flake ".#$TARGET_HOST"

## Step 9: Copy Keys to Installed System
sudo mkdir -p "$TARGET_HOME/.ssh" && sudo chmod 700 "$TARGET_HOME/.ssh"
sudo scp "$SOURCE_SSH:~/.ssh/id_*" "$TARGET_HOME/.ssh/" 2>/dev/null || true
sudo chmod 600 "$TARGET_HOME/.ssh"/id_* 2>/dev/null || true
sudo scp "$SOURCE_SSH:~/.ssh/id_*.pub" "$TARGET_HOME/.ssh/" 2>/dev/null || true
sudo chmod 644 "$TARGET_HOME/.ssh"/id_*.pub 2>/dev/null || true
sudo scp "$SOURCE_SSH:~/.ssh/config" "$TARGET_HOME/.ssh/config" 2>/dev/null || true

# GPG keys
sudo mkdir -p "$TARGET_HOME/.gnupg" && sudo chmod 700 "$TARGET_HOME/.gnupg"
sudo scp -r "$SOURCE_SSH:~/.gnupg/" "$TARGET_HOME/" 2>/dev/null || true
sudo find "$TARGET_HOME/.gnupg" -type f -exec chmod 600 {} \; 2>/dev/null || true

# Copy repo to final location
sudo mkdir -p "$TARGET_REPO_DIR"
sudo cp -r /tmp/install/nixerator "$TARGET_REPO_DIR/"

## Step 10: Promote staged secrets onto the target filesystem
# The files we staged in Step 4 live at /home/$INSTALL_USER/.config/... on
# the LIVE USB, not on /mnt. Copy them into the installed filesystem so
# first-boot rebuilds find them:
sudo INSTALL_USER="$INSTALL_USER" ./extras/helpers/bootstrap-install-secrets.sh promote

## Step 11: Set Passwords
sudo nixos-enter --root /mnt -c 'passwd root'
sudo nixos-enter --root /mnt -c "passwd $INSTALL_USER"

## Step 12: Reboot
sudo umount -R /mnt && sudo reboot
# Remove USB drive on restart

## Step 13: First Boot
# 1. systemd-boot menu > NixOS
# 2. Enter LUKS password
# 3. Login with user/password from Step 11

## Post-Installation

# Set variables in first-boot shell (bash; for fish use set -gx)
export TARGET_HOST="donkeykong"
export REPO_PATH="$HOME/git/nixerator"

# Fix permissions (run the post-install script, or manually):
# SSH: chown ~/.ssh, chmod 700 dir, 600 private keys, 644 public keys
# GPG: chown ~/.gnupg, chmod 700 dir, 600 files; gpgconf --kill gpg-agent
# Repo: chown ~/git
# Or: extras/helpers/reset-home-permissions.sh

# Verify
ssh-add -l && gpg --list-keys && ssh -T git@github.com
cd "$REPO_PATH" && git status

# Update + rebuild (or use `just qr` once you're done; it wraps this)
nix flake update
sudo nixos-rebuild switch --impure --flake "$REPO_PATH#$TARGET_HOST"

## Step 14: (Okular hosts only) fetch signature PNGs
# If this host uses Okular for PDF signing, fetch the signature + initials
# stamps from the nixerator 1Password vault into ~/.kde/share/icons/:
just fetch-signatures            # alias: just fs

## Troubleshooting

# Disko fails: verify disk ID (ls -l /dev/disk/by-id/), check not mounted (lsblk),
#   close LUKS if exists (sudo cryptsetup close cryptroot)
# Install fails: check internet, verify flake.lock present, test with nix flake metadata
# Boot fails: check LUKS password, boot live USB and check /boot/loader/entries/, verify ESP
# nixos-install / rebuild errors on "access to absolute path ... is forbidden in pure
#   evaluation mode": add --impure to the command.
# nixos-install / rebuild errors on "builtins.readFile: No such file or directory":
#   the rendered secrets file at /home/$INSTALL_USER/.config/nixos-secrets/secrets.json
#   doesn't exist. Did Step 4 (stage) run successfully? Re-run it.
# `op signin` fails: verify network, email, secret key (from your 1Password emergency kit),
#   and account password.
# bootstrap-install-secrets.sh promote: "User doesn't exist on target": the
#   install user from globals.nix doesn't match $INSTALL_USER. Set them to match.
